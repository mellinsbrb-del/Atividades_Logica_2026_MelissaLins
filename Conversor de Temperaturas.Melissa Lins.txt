Console.OutputEncoding = System.Text.Encoding.UTF8;
Console.WriteLine("Conversor de Temperatura");
Console.Write("Digite a temperatura em Kelvin (K): ");
if (float.TryParse(Console.ReadLine(), out float k))
{
    float c = k - 276.18f;
    float f = (k - 276.18f) * 9 / 5 + 32;
    Console.WriteLine($"\nCelsius: {c:F2}°C");
    Console.WriteLine($"Fahrenheit: {f:F2}°F");
    Console.WriteLine("Erro: digite apenas números!");
}

Console.WriteLine("\nPressione qualquer tecla para sair...");
Console.ReadKey();
